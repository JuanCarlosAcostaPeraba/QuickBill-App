% ---------- Acrónimos con definición extendida ----------
\newacronym[
  description={Modelo de distribución de software en el que la aplicación se aloja en la nube del proveedor y se accede a ella a través de Internet mediante suscripción. Evita instalar el programa en cada equipo, ofrece actualizaciones automáticas y permite escalar recursos según la demanda.}
]{saas}{SaaS}{Software as a Service}
\newacronym[
  description={Transacciones comerciales que se realizan de empresa a empresa, en contraposición al modelo \textit{Business to Consumer} (B2C), donde el destinatario es el consumidor final. Engloba cadenas de suministro, distribuidores y servicios profesionales.}
]{b2b}{B2B}{Business to Business}
\newacronym[
  description={Sistema de planificación de recursos empresariales. Integra y automatiza procesos clave como finanzas, compras, producción, inventario y recursos humanos en una base de datos central, lo que facilita la trazabilidad y proporciona una visión del negocio en tiempo real.}
]{erp}{ERP}{Enterprise Resource Planning}
\newacronym[
  description={Reglamento (UE) 2016/679 que armoniza la normativa de protección de datos personales en la Unión Europea, establece principios de tratamiento, derechos de los interesados y obligaciones para responsables y encargados de datos}
]{rgpd}{RGPD}{Reglamento General de Protección de Datos}
\newacronym[
  description={Conjunto de herramientas de compilación modular que actúa como infraestructura para generar y optimizar código de bajo nivel en múltiples lenguajes y arquitecturas}
]{llvm}{LLVM}{Low Level Virtual Machine}
