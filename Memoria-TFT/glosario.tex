% ---------- Glosario ----------
\newglossaryentry{saas}{
  name={SaaS},
  description={Modelo \emph{Software as a Service}: la aplicación se aloja en la nube del proveedor y se consume vía Internet mediante suscripción. Evita instalaciones locales, actualiza de forma automática y escala según la demanda.},
  sort=saas
}
\newglossaryentry{b2b}{
  name={B2B},
  description={Siglas de \emph{Business to Business}. Transacciones comerciales de empresa a empresa, a diferencia del modelo B2C dirigido al consumidor final. Incluye cadenas de suministro, distribuidores y servicios profesionales.},
  sort=b2b
}
\newglossaryentry{erp}{
  name={ERP},
  description={\emph{Enterprise Resource Planning}. Sistema que integra y automatiza procesos empresariales (finanzas, compras, producción, inventario, RR.~HH.) en una base de datos central para ofrecer trazabilidad y visión en tiempo real.},
  sort=erp
}
\newglossaryentry{rgpd}{
  name={RGPD},
  description={Reglamento (UE) 2016/679 de protección de datos personales que establece principios de tratamiento, derechos de los interesados y obligaciones para responsables y encargados.},
  sort=rgpd
}
\newglossaryentry{llvm}{
  name={LLVM},
  description={Conjunto de herramientas de compilación modular que actúa como infraestructura para generar y optimizar código de bajo nivel en múltiples lenguajes y arquitecturas.},
  sort=llvm
}
\newglossaryentry{astro}{
  name={Astro},
  description={Framework web basado en la arquitectura de islas. Genera HTML estático en el servidor y añade JavaScript solo en los componentes que lo necesitan, lo que mejora el rendimiento y la accesibilidad},
  sort=astro
}
